CMAKE_MINIMUM_REQUIRED(VERSION 3.13)
PROJECT(toolkit-common VERSION 2.1.0 LANGUAGES ${LANG})

EXECUTE_PROCESS(COMMAND date +"%Y%m%d%H%I" OUTPUT_VARIABLE toolkit-common_VERSION_BUILD)
SET(toolkit-common_VERSION_STAGE "beta")

if(NOT LANG)
	MESSAGE("Use la variable LANG para indicar el lenguaje.")	
elseif(${LANG} STREQUAL "CXX")
	MESSAGE("Generando componetes en C++ ...")	
	INCLUDE(CMakeLists-CXX)
elseif(${LANG} STREQUAL "C")
	MESSAGE("Generando componetes en C ...")	
	INCLUDE(CMakeLists-C)
elseif(${LANG} STREQUAL "F")
	MESSAGE("Generando componetes en Fortran ...")	
	INCLUDE(CMakeLists-F)
elseif(${LANG} STREQUAL "OBJC")
	MESSAGE("Generando componetes en Objective-C ...")	
	INCLUDE(CMakeLists-OBJC)
elseif(${LANG} STREQUAL "JAVA")
	MESSAGE("Generando componetes en Java ...")	
	INCLUDE(CMakeLists-JAVA)
elseif(${LANG} STREQUAL "PY")
	MESSAGE("Generando componetes en Python ...")	
	INCLUDE(CMakeLists-PY)
elseif()
	MESSAGE(FATAL_ERROR "El valor de LANG es desconocido.")
endif()

INCLUDE (InstallRequiredSystemLibraries)
#SET (CPACK_RESOURCE_FILE_LICENSE "${CMAKE_CURRENT_SOURCE_DIR}/License.txt")
SET (CPACK_PACKAGE_VERSION_MAJOR "${toolkit-common_VERSION_MAJOR}")
SET (CPACK_PACKAGE_VERSION_MINOR "${toolkit-common_VERSION_MINOR}")
SET (CPACK_PACKAGE_VERSION_PATCH "${toolkit-common_VERSION_PATCH}")
INCLUDE (CPack)
