CMAKE_MINIMUM_REQUIRED(VERSION 3.0)
PROJECT(toolkit-cliendb VERSION 4.0.0 LANGUAGES ${LANG})

IF(NOT LANG)
	MESSAGE(FATAL_ERROR "Use la variable LANG para indicar el lenguaje.")	
ELSEIF(${LANG} STREQUAL "CXX")
	MESSAGE("El Lenguaje seleccionado es C++ ...")
	INCLUDE(CXX.cmake)
ELSEIF(${LANG} STREQUAL "Java")
	MESSAGE("El Lenguaje seleccionado es Java ...")
	INCLUDE(Java.cmake)
ENDIF()

INCLUDE (InstallRequiredSystemLibraries)
SET(CPACK_PACKAGE_VERSION_MAJOR "${toolkit-cliendb_VERSION_MAJOR}")
SET(CPACK_PACKAGE_VERSION_MINOR "${toolkit-cliendb_VERSION_MINOR}")
SET(CPACK_PACKAGE_VERSION_PATCH "${toolkit-cliendb_VERSION_PATCH}")
include (CPack)
