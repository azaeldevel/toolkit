CMAKE_MINIMUM_REQUIRED(VERSION 3.0)
PROJECT(octetos-toolkit-clientdb VERSION 5.0.0 LANGUAGES ${LANG})
EXECUTE_PROCESS(COMMAND date +"%Y%m%d%H%I" OUTPUT_VARIABLE octetos-toolkit-clientdb_VERSION_BUILD)

IF(NOT LANG)
	MESSAGE(FATAL_ERROR "Use la variable LANG para indicar el lenguaje.")	
ELSEIF(${LANG} STREQUAL "CXX")
	MESSAGE("El Lenguaje seleccionado es C++ ...")
	INCLUDE(CMakeLists-CXX.cmake)
ENDIF()

INCLUDE (InstallRequiredSystemLibraries)
SET (CPACK_PACKAGE_VERSION_MAJOR "${octetos-toolkit-clientdb_VERSION_MAJOR}")
SET (CPACK_PACKAGE_VERSION_MINOR "${octetos-toolkit-clientdb_VERSION_MINOR}")
SET (CPACK_PACKAGE_VERSION_PATCH "${octetos-toolkit-clientdb_VERSION_PATCH}")
INCLUDE (CPack)
